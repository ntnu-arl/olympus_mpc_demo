84 num_legs(state_struct.num_legs),

88 this->DeclareVectorInputPort("full_state", drake::systems::BasicVector<double>(num_states));
89
90 pos_ids.push_back(0);
91 vel_ids.push_back(num_pos_states);

98 int leg_vel_start_index = num_base_vel + (leg_id * num_leg_joints) + num_pos_states;

101 vel_ids.push_back(leg_vel_start_index);

105 //1. Declare an output port for each leg's state
106 for (int i = 0; i < num_legs; ++i) {

108 leg_prefix[i]+"_leg_state",
109 drake::systems::BasicVector<double>(num_leg_joints * 2),
110 std::bind(&olympus_state_demultiplexer::CalcLegStateOutput, this, std::placeholders::_1, std::placeholders::_2, i));
111 }
112
113 //2. Declare an output port for the body orientation
114 this->DeclareVectorOutputPort(
115 "olympus_body_state",
116 drake::systems::BasicVector<double>(13), //output full body state always
117 std::bind(&olympus_state_demultiplexer::CalcBodyStateOutput, this, std::placeholders::_1, std::placeholders::_2));

121void olympus_state_demultiplexer::CalcBodyStateOutput (
122 const drake::systems::Context<double>& context,
123 drake::systems::BasicVector<double>* output ) const {
124
125 const auto& full_state = this->get_input_port(0).Eval(context);
126 auto output_vector = output->get_mutable_value();
127
128 Eigen::Quaterniond quat;
129 Eigen::Quaterniond quat_off;
130 Eigen::Quaterniond quat_mocap;
131
132 const double * angle;
133 const double * twist;
134
135 output_vector.setZero();
136 switch (base_joint_type){
137 case olympus_base_joint::weld:
138 break;
139
140 case olympus_base_joint::floating:
141 output_vector.head(4) = full_state.segment(pos_ids[0], 4);
142 output_vector.segment(7, 3) = full_state.segment(vel_ids[0], 3);
143 break;
144
145 case olympus_base_joint::revolute_x:
146 angle = &full_state[ pos_ids[0] ];
147 twist = &full_state[ vel_ids[0] ];
148
149 quat = Eigen::Quaterniond( Eigen::AngleAxisd( *angle, Eigen::Vector3d(1,0,0)) );
150 quat_off = Eigen::Quaterniond( Eigen::AngleAxisd(-M_PI_2, Eigen::Vector3d(0,1,0)) );
151 quat_mocap = quat_off*quat ;
152
153 output_vector[0] = quat_mocap.w();
154 output_vector.segment(1, 3) = quat_mocap.vec();
155 output_vector[9] = *twist; //z-axis
156 break;
157
158 case olympus_base_joint::revolute_y:
159 angle = &full_state[ pos_ids[0] ];
160 twist = &full_state[ vel_ids[0] ];
161
162 quat = Eigen::Quaterniond( Eigen::AngleAxisd( *angle, Eigen::Vector3d(0,1,0)) );
163 quat_off = Eigen::Quaterniond( Eigen::AngleAxisd(-M_PI_2, Eigen::Vector3d(1,0,0)) );
164 quat_mocap = quat_off*quat ;
165
166 output_vector[0] = quat_mocap.w();
167 output_vector.segment(1, 3) = quat_mocap.vec();
168 output_vector[9] = *twist; //z-axis
169 break;
170
171 case olympus_base_joint::revolute_z:
172 angle = &full_state[ pos_ids[0] ];
173 twist = &full_state[ vel_ids[0] ];
174
175 quat = Eigen::Quaterniond( Eigen::AngleAxisd( *angle, Eigen::Vector3d(0,0,1)) ) ;
176
177 output_vector[0] = quat.w();
178 output_vector.segment(1, 3) = quat.vec();
179 output_vector[9] = *twist; //z-axis
180
181 }
182}

184void olympus_state_demultiplexer::CalcLegStateOutput(
185 const drake::systems::Context<double>& context,
186 drake::systems::BasicVector<double>* output,
187 const int & leg_index) const {
188
189 const auto& full_state = this->get_input_port(0).Eval(context);
190 auto output_vector = output->get_mutable_value();
191
192 //index 0 in `pos_ids/vel_ids` is for body states
193 const int & leg_pos_start_index = pos_ids[leg_index + 1];
194 const int & leg_vel_start_index = vel_ids[leg_index + 1];
195
196 // Copy the leg's q (positions) and v (velocities) to the output vector
197 output_vector.head(num_leg_joints) = full_state.segment(leg_pos_start_index, num_leg_joints);
198 output_vector.tail(num_leg_joints) = full_state.segment(leg_vel_start_index, num_leg_joints);
199}

203leg_system::leg_system(drake_builder & builder, drake_plant* plant_,
204 const std::string & leg_prefix_)
205 : plant(plant_),

207
208 YAML::Node config = YAML::LoadFile("config/olympus_param.yaml");
209
210 //1. Get joint names:
211 std::vector<std::string> joint_names;
212 if (!config["joint_names"]) {
213 std::cerr << "[olympus drake]: 'joint_names' key not found in the YAML file." << std::endl;
214 } else {
215 if (config["joint_names"][leg_prefix]) {
216 for (const auto& joint : config["joint_names"][leg_prefix]) {
217 joint_names.push_back(joint.as<std::string>());
218 }
219 } else {
220 std::cerr << "[olympus drake] Error: '" << leg_prefix << "' key not found in 'joint_names'." << std::endl;
221 }
222 }
223
224 //2. Get body names:
225 std::vector<std::string> body_names;
226 if (!config["body_names"]) {
227 std::cerr << "[olympus drake]: 'body_names' key not found in the YAML file." << std::endl;
228 } else {
229 if (config["body_names"][leg_prefix]) {
230 for (const auto& body : config["body_names"][leg_prefix]) {
231 body_names.push_back(body.as<std::string>());
232 }
233 } else {
234 std::cerr << "[olympus drake] Error: '" << leg_prefix << "' key not found in 'body_names'." << std::endl;
235 }
236 }
237
238
239 //3. Get joint and body object pointers from loaded urdf
240 for (int i=0;i<num_joints;i++){
241 leg_bodies[i] = &plant->GetBodyByName(body_names[i]);
242 leg_joints[i] = &plant->GetJointByName(joint_names[i]);
243 }
244 set_state_ids(); //TODO: see if it is needed
245
246 bool add_spring = true;
247 //4. Add other parameters, such as springs etc.
248 if (!config["add_leg_spring"]) {
249 std::cerr << "[olympus drake]: 'add_leg_spring' key not found in the YAML file." << std::endl;
250 std::cerr << "[olympus drake]: By default the spring is added" << std::endl;
251 }else{
252 add_spring = config["add_leg_spring"].as<bool>();
253 }
254
255 std::vector<double> motor_limit_;
256 if (!config["motor_limits"]) {
257 std::cerr << "[olympus drake]: 'motor_limits' key not found in the YAML file." << std::endl;
258 }else{
259 motor_limit_ = config["motor_limits"].as<std::vector<double>>();
260 if (motor_limit_.size()!= 3){
261 std::cerr << "[olympus drake]: 3 values for motor_limits must be provided. Using default limits of 20Nm" << std::endl;
262 }else {
263 for (int i=0; i<3;i++){ motor_limit[i] = motor_limit_[i];}
264 }
265
266 }
267
268 //4. Complete leg system
269 if (add_spring){ add_linear_spring(); }
270 add_bushing_joint();
271 add_actuators();
272 add_PID_system(builder);
273}

381void leg_system::add_PID_system(drake_builder & builder){
382 //1. Define Control Projection Matrix:
383 //Given estimated state x_in = (q_in, v_in), the controlled state x_c = (q_c, v_c) is computed by x_c = P_x * x_in
384 int num_states = num_joints*2;
385
386 //2. Populate it
387 Eigen::MatrixXd ControlProjectionMatrix;
388 ControlProjectionMatrix.resize(6,num_states);
389 ControlProjectionMatrix.setZero();
390 for ( int i_s=0;i_s<6;i_s++){
391 ControlProjectionMatrix(i_s,state_ids[i_s]) = 1;
392 }
393
394 //3. Add PID system
395 controller = builder.AddNamedSystem<drake::systems::controllers::PidController<double>>(
396 leg_prefix+"_controller",
397 ControlProjectionMatrix,
398 Gains.Kp,
399 Gains.Ki,
400 Gains.Kd);
401
402 //Motor limits must be enforced to PID system independently, as actuator limits are efnorces only if there are joint pd.
403 Eigen::Vector3d input_limit(motor_limit[0],motor_limit[1],motor_limit[2]);
404 controller_saturation = builder.AddSystem<drake::systems::Saturation<double>>(-input_limit,input_limit);
405 builder.Connect( controller -> get_output_port(), controller_saturation -> get_input_port() );
406
407
408 controller_desired_state_port = builder.ExportInput(controller -> get_input_port_desired_state(),leg_prefix+"_leg_setpoint");
409}

457

461

472

474 if (!robot_config["testbed_params"]) {

476 }else{
477 switch(joint_type){
478 case olympus_base_joint::revolute_x:
479 testbed_DZ = robot_config["testbed_params"]["roll_DX"].as<double>();
480 testbed_damping = robot_config["testbed_params"]["damping_roll"].as<double>();
481 break;
482 case olympus_base_joint::revolute_y:
483 testbed_DZ = robot_config["testbed_params"]["pitch_DX"].as<double>();
484 testbed_damping = robot_config["testbed_params"]["damping_pitch"].as<double>();
485 break;
486 case olympus_base_joint::revolute_z:
487 testbed_DZ = robot_config["testbed_params"]["yaw_DX"].as<double>();
488 testbed_damping = robot_config["testbed_params"]["damping_yaw"].as<double>();
489 break;
490 }
491 }
492 }
493
494 if (!robot_config["torso_name"]) {
495 std::cerr << "[olympus drake]: 'torso_name' key not found in the YAML file. Using "<< torso_name << std::endl;
496 }else{
497 torso_name = robot_config["torso_name"].as<std::string>();
498 }
499
500 std::vector<std::string> leg_prefix_;
501 for (auto leg:robot_config["joint_names"]){
502 leg_prefix_.push_back( leg.first.as<std::string>() );
503 }
504 if (leg_prefix_.size()==4){
505 std::copy(leg_prefix_.begin(),leg_prefix_.end(),leg_prefix.begin());
506 }else{
507 std::cerr << "[olympus drake]: More joint groupds were loaded than expected. These are: "<<std::endl;
508 for (auto str:leg_prefix_){ std::cerr << str <<std::endl;}
509 }
510
511 #pragma endregion
512
513 //B. add systems
514 #pragma region
515 int actuators_per_leg = 3;
516 int num_legs = 4;
517 int num_actuators = num_legs*actuators_per_leg;
518 //1. Add quadruped plant
519 auto [plant_,scene_graph] = drake::multibody::AddMultibodyPlantSceneGraph(&builder,time_step);
520 plant = &plant_;
521
522 drake::multibody::Parser parser(plant);
523 quadruped = parser.AddModels("urdf/olympus_drake.urdf").at(0);
524 plant->RenameModelInstance(quadruped,robot_name);
525 get_joint_mapping(plant->GetJointIndices(quadruped),robot_config);
526
527 add_base_joint( plant->GetBodyByName(torso_name) );
528
529 for (int i=0;i<num_legs;i++){
530 legs[i] = new leg_system(builder, plant, leg_prefix[i]);
531 legs[i]->handle_shanks_collisions(scene_graph);
532 }
533
534 //2. Set plant simulation parameters
535 plant->mutable_gravity_field().set_gravity_vector(custom_gravity);
536 if (joint_type == olympus_base_joint::floating){
537 std::cout << "[drake simulation]: With floating joint, gravity is always set to zero" <<std::endl;
538 plant->mutable_gravity_field().set_gravity_vector(Eigen::Vector3d().Zero());
539 }
540 plant->set_discrete_contact_approximation(drake::multibody::DiscreteContactApproximation::kSap);
541 plant->Finalize();
542
543 //3. Add rest of systems in the simulation (PID, Multiplexer, "Demultiplexer")
544 state_info_struct state_info(joint_type);
545
546 auto mux = builder.AddSystem<drake::systems::Multiplexer>(std::vector<int>(num_legs, actuators_per_leg));
547 auto demux = builder.AddSystem<drake::systems::Demultiplexer>(num_actuators,actuators_per_leg);
548 outputer = builder.AddSystem<olympus_state_demultiplexer>(state_info);
549 outputer->set_name("olympus_state_demultiplexer");
550
551 #pragma endregion
552
553 //C. connect systems
554 builder.Connect(mux->get_output_port(),plant->get_actuation_input_port());
555 builder.Connect(plant->get_state_output_port(),outputer->get_input_port());
556 builder.Connect(plant->get_net_actuation_output_port(quadruped),demux->get_input_port());
557
558 //Add "encoders"

607void olympus_plant::add_base_joint( const drake_rigidBody & torso_body ){
608
609 const drake::multibody::Joint< double >* world_joint_ptr =nullptr;
610 const drake_rigidBody* testbed_body_ptr =nullptr;
611
612 switch (joint_type){
613 case olympus_base_joint::revolute_x :
614 testbed_body_ptr = add_testbed_body();
615
616 world_joint_ptr = & plant->AddJoint<drake::multibody::RevoluteJoint>(
617 "world_joint",
618 plant->world_body(),
619 drake_tfd(Eigen::Translation3d{0,0,1}),
620 *testbed_body_ptr,
621 drake_tfd(),
622 Eigen::Vector3d{0,0,1}, // z-axis
623 testbed_damping
624 );
625
626 joint_ptr = & plant->AddJoint< drake::multibody::WeldJoint >(
627 "body_joint",
628 *testbed_body_ptr,
629 drake_tfd(drake::math::RollPitchYawd(0,-M_PI_2,0), drake::Vector3<double>(testbed_DZ,0,0)), //z-translation
630 torso_body,
631 drake_tfd(),
632 drake_tfd::Identity()
633 );
634
635 break;
636
637 case olympus_base_joint::revolute_y :
638 testbed_body_ptr = add_testbed_body();
639
640 world_joint_ptr = & plant->AddJoint<drake::multibody::RevoluteJoint>(
641 "world_joint",
642 plant->world_body(),
643 drake_tfd(Eigen::Translation3d{0,0,1}),
644 *testbed_body_ptr,
645 drake_tfd(),
646 Eigen::Vector3d{0,0,1}, // z-axis
647 testbed_damping
648 );
649
650 joint_ptr = & plant->AddJoint< drake::multibody::WeldJoint >(
651 "body_joint",
652 *testbed_body_ptr,
653 drake_tfd(drake::math::RollPitchYawd(-M_PI_2,0,0), drake::Vector3<double>(0,testbed_DZ,0)), //z-translation
654 torso_body,
655 drake_tfd(),
656 drake_tfd::Identity()
657 );
658 break;
659
660 case olympus_base_joint::revolute_z :
661 testbed_body_ptr = add_testbed_body();
662
663 world_joint_ptr = & plant->AddJoint<drake::multibody::RevoluteJoint>(
664 "world_joint",
665 plant->world_body(),
666 drake_tfd(Eigen::Translation3d{0,0,1}),
667 *testbed_body_ptr,
668 drake_tfd(),
669 Eigen::Vector3d{0,0,1}, // z-axis
670 testbed_damping

674 "body_joint",
675 *testbed_body_ptr,
676 drake_tfd( Eigen::Translation3d{0,0,testbed_DZ}), //z-translation
677 torso_body,
678 drake_tfd(),

691 break;

693 default: //weld
694 joint_ptr = & plant->AddJoint< drake::multibody::WeldJoint >(
695 "body_joint",
696 plant->world_body(),

714void olympus_plant::get_joint_mapping(
715 const std::vector<drake::multibody::JointIndex> & joint_index_vector,
716 const YAML::Node & robot_config){
717
718 //1. Get a joint(first) of each leg system and add it in the `elements` vector
719 std::vector<std::string> joint_elements;
720 for (auto str:leg_prefix){
721 joint_elements.push_back(robot_config["joint_names"][str].as<std::vector<std::string>>()[0] );
722 }
723
724 //2. Get the `joint_index` of the first joint of each leg
725 std::vector<int> joint_elements_ids;
726 for( auto joint_elem:joint_elements){
727 auto& joint = plant->GetJointByName(joint_elem,quadruped);
728
729 // std::cout<< int( joint.index() ) <<std::endl;
730 // std::cout<< joint_elem <<std::endl;
731
732 joint_elements_ids.push_back( int( joint.index() ) );
733 }
734
735 //3. Map urdf leg system order with configuration order
736 auto ascending_ind = ascending_indices<int>(joint_elements_ids);
737
738 //4. Map configuration order with urdf order (to get states). (Its like inversing)
739 auto map_ind_ = ascending_indices<int>(ascending_ind);
740 std::copy(map_ind_.begin(),map_ind_.end(),leg_sim_order.begin());
741}
